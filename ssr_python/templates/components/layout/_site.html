{# ============================================================================ SITE COMPONENT — Transparent wrapper for multipage support Passes through to render its children (pages). When site-level theme fonts exist, wraps output in a site-wrapper div that carries font CSS variables for inheritance. {% endif %} {# Render site-level header components shared across pages #} {% for child in component.header | default([]) %} {{ render_component(child, tokens, path + ['header', loop.index0]) }} {% endfor %} {# Render pages #} {% for child in component.components | default([]) %} {{ render_component(child, tokens, path + ['components', loop.index0]) }} {% endfor %} {# Render site-level footer components shared across pages #} {% for child in component.footer | default([]) %} {{ render_component(child, tokens, path + ['footer', loop.index0]) }} {% endfor %} {% if has_theme_fonts or has_header or has_footer %}
{% endif %} {% endmacro %}
